34/// ===================== Syntax Analyzer =====================
35pub struct LolcodeCompiler {
36    /// lexer goes char by char and tokenizes input, also validations tokens and chars are allowed
37    lexer: LolcodeLexicalAnalyzer,
38    /// current token being pasered 
39    current_tok: String,
40    /// syntax tree that gets generated is the output of syntax analyzer
41    pub tree: Vec<Node>,
42    /// used to close open tags
43    stack: Vec<Vec<Node>>,
44    /// Holds a pre-consumed dispatch token (#maek / #gimmeh) so it lands
45    /// as the first string inside the sub-list that owns it.
46    pending: Option<String>,
47}
48
49impl LolcodeCompiler {
50    /// constructor for syntax analyzer
51    pub fn new() -> Self {
52        Self {
53            lexer: LolcodeLexicalAnalyzer::new(""),
54            current_tok: String::new(),
55            tree: Vec::new(),
56            stack: Vec::new(),
57            pending: None,
58        }
59    }
60
61    /// new scope block is created push close tag to stack
62    fn open_list(&mut self) {
63        self.stack.push(Vec::new());
64        // If a dispatch token (#maek / #gimmeh) was saved, make it the first
65        // string in this new scope.
66        if let Some(tok) = self.pending.take() {
67            self.add_str(tok);
68        }
69    }
70    /// adds string to syntax tree
71    fn add_str(&mut self, s: String) {
72        if let Some(top) = self.stack.last_mut() {
73            top.push(Node::Str(s));
74        }
75    }
76    /// close scope block 
77    fn close_list(&mut self) {
78        if let Some(finished) = self.stack.pop() {
79            let node = Node::List(finished);
80            if let Some(parent) = self.stack.last_mut() {
81                parent.push(node);
82            } else {
83                self.tree.push(node);
84            }
85        }
86    }
87
88    /// sets first current token  
89    fn start(&mut self) {
90        //gets possible token
91        let candidate = self.lexer.tokens.pop().unwrap_or_default();
92        //checks if token is valid
93        if self.lexer.lookup(&candidate) {
94            // sets current token
95            self.current_tok = candidate;
96
97        // errors if token is not allowed
98        } else if !candidate.is_empty() {
99            eprintln!("Lexical error: '{}' is not a recognized token.", candidate);
100            std::process::exit(1);
101        // errors if token does not exist
102        } else {
103            eprintln!("Error: the provided source is empty.");
104            std::process::exit(1);
105        }
106    }
107

159/// ===================== Compiler Trait Impl =====================
160impl Compiler for LolcodeCompiler {
161
162    /// constructor - sets lexer starts tokenizing input and first sets current token
163    fn compile(&mut self, source: &str) {
164        self.lexer = LolcodeLexicalAnalyzer::new(source);
165        self.lexer.tokenize();
166        self.start();
167    }
168
169    /// gets next token
170    fn next_token(&mut self) -> String {
171        // gets potential token
172        let candidate = self.lexer.tokens.pop().unwrap_or_default();
173        // looks up candidate token
174        if self.lexer.lookup(&candidate) {
175            // if token set current token and return it
176            self.current_tok = candidate.clone();
177            candidate
178        } else if candidate.is_empty() {
179            // if empty clears token and returns empty string
180            self.current_tok.clear();
181            String::new()
182        } else {
183            // if neither retirn error since token is not valid
184            eprintln!("Lexical error: '{}' is not a recognized token.", candidate);
185            std::process::exit(1);
186        }
187    }
188
189    /// parse input
190    fn parse(&mut self) {
191        self.parse_lolcode();
192    }
193    /// gets current token
194    fn current_token(&self) -> String {
195        self.current_tok.clone()
196    }
197    /// sets current token
198    fn set_current_token(&mut self, tok: String) {
199        self.current_tok = tok;
200    }
201}

209    /// <lolcode> ::= HAI <comments> <head> <body> KBYE
210    fn parse_lolcode(&mut self) {
211        self.open_list();
212        if self.is_hai(&self.current_tok) {
213            self.add_str(self.current_tok.clone());
214            self.next_token();
215        } else {
216            eprintln!("Syntax error: '{}' was found when '#hai' was expected.", self.current_tok);
217            std::process::exit(1);
218        }
219
220        self.parse_comment();
221
222        // multiple tokens can be expected after #maek need to account for head, paragraf and list
223        if self.is_maek(&self.current_tok) {
224            self.pending = Some(self.current_tok.clone());
225            self.next_token();
226            if self.is_head(&self.current_tok) {
227                self.parse_head();
228            } else if self.is_paragraf(&self.current_tok) {
229                self.parse_paragraph();
230            } else if self.is_list(&self.current_tok) {
231                self.parse_list();
232            } else {
233                eprintln!("Syntax error: '{}' was found when 'head', 'paragraf', or 'list' was expected after '#maek'.", self.current_tok);
234                std::process::exit(1);
235            }
236        }
237
238        self.parse_body();
239
240        if self.is_kbye(&self.current_tok) {
241            self.add_str(self.current_tok.clone());
242            self.next_token();
243        } else {
244            eprintln!("Syntax error: '{}' was found when '#kbye' was expected.", self.current_tok);
245            std::process::exit(1);
246        }
247        self.close_list();
248    }

305    /// <comments> ::= <comment> <comments> | empty
306    /// <comment>  ::= OBTW TEXT TLDR
307    fn parse_comment(&mut self) {
308        while self.is_obtw(&self.current_tok) {
309            self.open_list();
310            self.add_str(self.current_tok.clone());
311            self.next_token();
312            self.parse_text();
313            if self.is_tldr(&self.current_tok) {
314                self.add_str(self.current_tok.clone());
315                self.next_token();
316            } else {
317                eprintln!("Syntax error: '{}' was found when '#tldr' was expected.", self.current_tok);
318                std::process::exit(1);
319            }
320            self.close_list();
321        }
322    }

324    /// <body> ::= <inner-body> <body> | empty
325    /// <inner-body> ::= <paragraph>
326    ///     | <comment>
327    ///     | <bold>
328    ///     | <italics>
329    ///     | <list>
330    ///     | <link>
331    ///     | <newline>
332    ///     | <variable-define>
333    ///     | <variable-use>
334    ///     | TEXT
335    fn parse_body(&mut self) {
336        while !self.is_kbye(&self.current_tok) && !self.current_tok.is_empty() {
337            if self.is_obtw(&self.current_tok) {
338                self.parse_comment();
339            } else if self.is_ihaz(&self.current_tok) {
340                self.parse_variable_define();
341            } else if self.is_lemmesee(&self.current_tok) {
342                self.parse_variable_use();
343            } else if self.is_maek(&self.current_tok) {
344                self.pending = Some(self.current_tok.clone());
345                self.next_token();
346                if self.is_paragraf(&self.current_tok) {
347                    self.parse_paragraph();
348                } else if self.is_list(&self.current_tok) {
349                    self.parse_list();
350                } else {
351                    eprintln!("Syntax error: '{}' was found when 'paragraf' or 'list' was expected after '#maek'.", self.current_tok);
352                    std::process::exit(1);
353                }
354            } else if self.is_gimmeh(&self.current_tok) {
355                self.pending = Some(self.current_tok.clone());
356                self.next_token();
357                if self.is_bold(&self.current_tok) {
358                    self.parse_bold();
359                } else if self.is_italics(&self.current_tok) {
360                    self.parse_italics();
361                } else if self.is_linx(&self.current_tok) {
362                    self.parse_link();
363                } else if self.is_newline(&self.current_tok) {
364                    self.parse_newline();
365                } else {
366                    eprintln!("Syntax error: '{}' was found after '#gimmeh' in body.", self.current_tok);
367                    std::process::exit(1);
368                }
369            } else if self.is_text(&self.current_tok) {
370                self.parse_text();
371            } else {
372                eprintln!("Syntax error: unexpected token '{}' in body.", self.current_tok);
373                std::process::exit(1);
374            }
375        }
376    }

378    /// <paragraph> ::= PARAGRAF <variable-define> <inner-paragraph> MKAY
379    /// Called with current_tok == "paragraf" (caller already consumed #maek).
380    /// <inner-paragraph> ::= <inner-text> <inner-paragraph> | empty
381    /// <inner-text> ::= <variable-use>
382    ///     | <bold>
383    ///     | <italics>

394        } else {

396            std::process::exit(1);
397        }
398
399        self.parse_variable_define();
400        self.parse_inner_paragraph();
401
402        if self.is_mkay(&self.current_tok) {
403            self.add_str(self.current_tok.clone());
404            self.next_token();
405        } else {
406            eprintln!("Syntax error: '{}' was found when '#mkay' was expected.", self.current_tok);
407            std::process::exit(1);
408        }
409        self.close_list();
410    }

412    /// <inner-paragraph> ::= <inner-text> <inner-paragraph> | empty
413    fn parse_inner_paragraph(&mut self) {

420    fn parse_inner_text(&mut self) {
421        if self.is_mkay(&self.current_tok) || self.current_tok.is_empty() {
422            // empty
423        } else if self.is_lemmesee(&self.current_tok) {

425        } else if self.is_ihaz(&self.current_tok) {

427        } else if self.is_obtw(&self.current_tok) {
428            self.parse_comment();
429        } else if self.is_maek(&self.current_tok) {
430            self.pending = Some(self.current_tok.clone());
431            self.next_token();
432            if self.is_list(&self.current_tok) {
433                self.parse_list();
434            } else {

436                std::process::exit(1);
437            }
438        } else if self.is_gimmeh(&self.current_tok) {
439            self.pending = Some(self.current_tok.clone());
440            self.next_token();
441            if self.is_bold(&self.current_tok) {
442                self.parse_bold();
443            } else if self.is_italics(&self.current_tok) {

445            } else if self.is_linx(&self.current_tok) {
446                self.parse_link();
447            } else if self.is_newline(&self.current_tok) {
448                self.parse_newline();
449            } else {
450                eprintln!("Syntax error: '{}' was found after '#gimmeh' inside paragraph.", self.current_tok);

452            }
453        } else if self.is_text(&self.current_tok) {
454            self.parse_text();
455        } else {
456            eprintln!("Syntax error: unexpected token '{}' inside paragraph.", self.current_tok);

461    /// <variable-define> ::= IHAZ VARNAME ITIZ VARVALUE MKAY | empty

464            return; // empty
465        }
466
467        self.open_list();
468        self.add_str(self.current_tok.clone()); // #ihaz
469        self.next_token();
470
471        // VARNAME ::= any single word (A–Z, a–z, no spaces)
472        if !self.is_var_name(&self.current_tok) {
473            eprintln!("Syntax error: '{}' was found when a variable name (letters only) was expected.", self.current_tok);
474            std::process::exit(1);
475        }
476        self.add_str(self.current_tok.clone()); // VARNAME
477        self.next_token();
478
479        if self.is_itiz(&self.current_tok) {
480            self.add_str(self.current_tok.clone());
481            self.next_token();
482        } else {
483            eprintln!("Syntax error: '{}' was found when '#itiz' was expected.", self.current_tok);
484            std::process::exit(1);
485        }
486
487        // VARVALUE ::= allowed text characters
488        if !self.is_var_value(&self.current_tok) {
489            eprintln!("Syntax error: '{}' was found when a variable value was expected.", self.current_tok);
490            std::process::exit(1);
491        }
492        self.add_str(self.current_tok.clone()); // VARVALUE
493        self.next_token();
494
495        if self.is_mkay(&self.current_tok) {
496            self.add_str(self.current_tok.clone());
497            self.next_token();
498        } else {
499            eprintln!("Syntax error: '{}' was found when '#mkay' was expected.", self.current_tok);
500            std::process::exit(1);
501        }

503    }

564            self.next_token();
565        } else {
566            eprintln!("Syntax error: '{}' was found when 'italics' was expected.", self.current_tok);
567            std::process::exit(1);
568        }
569
570        self.parse_text();
571
572        if self.is_oic(&self.current_tok) {
573            self.add_str(self.current_tok.clone());
574            self.next_token();
575        } else {
576            eprintln!("Syntax error: '{}' was found when '#oic' was expected.", self.current_tok);
577            std::process::exit(1);
578        }
579        self.close_list();
580    }
581
582    /// <list> ::= LIST <list-items> MKAY
583    /// Called with current_tok == "list" (caller already consumed #maek).
584    fn parse_list(&mut self) {
585        self.open_list();
586        if self.is_list(&self.current_tok) {
587            self.add_str(self.current_tok.clone());
588            self.next_token();
589        } else {
590            eprintln!("Syntax error: '{}' was found when 'list' was expected.", self.current_tok);
591            std::process::exit(1);
592        }
593
594        self.parse_list_items();
595
596        if self.is_mkay(&self.current_tok) {
597            self.add_str(self.current_tok.clone());
598            self.next_token();
599        } else {
600            eprintln!("Syntax error: '{}' was found when '#mkay' was expected.", self.current_tok);
601            std::process::exit(1);
602        }
603        self.close_list();
604    }
605
606    /// <list-items> ::= <list-item> <list-items> | empty
607    fn parse_list_items(&mut self) {
608        while self.is_gimmeh(&self.current_tok) {
609            self.parse_inner_list();
610        }
611    }
612
613    /// <list-item> ::= GIMMEH ITEM <inner-list> OIC
614    /// Called with current_tok == "#gimmeh" (NOT pre-consumed).
615    /// <inner-list> ::= <bold> | <italics> | <link> | TEXT | <variable-use> | empty
616    fn parse_inner_list(&mut self) {
617        self.open_list();
618        if self.is_gimmeh(&self.current_tok) {
619            self.add_str(self.current_tok.clone());
620            self.next_token();
621        } else {
622            eprintln!("Syntax error: '{}' was found when '#gimmeh' was expected.", self.current_tok);
623            std::process::exit(1);
624        }
625
626        if self.is_item(&self.current_tok) {
627            self.add_str(self.current_tok.clone());
628            self.next_token();
629        } else {
630            eprintln!("Syntax error: '{}' was found when 'item' was expected.", self.current_tok);
631            std::process::exit(1);
632        }
633
634        if self.is_gimmeh(&self.current_tok) {
635            self.pending = Some(self.current_tok.clone());
636            self.next_token();
637            if self.is_bold(&self.current_tok) {
638                self.parse_bold();
639            } else if self.is_italics(&self.current_tok) {
640                self.parse_italics();
641            } else if self.is_linx(&self.current_tok) {
642                self.parse_link();
643            } else {
644                eprintln!("Syntax error: '{}' was found after '#gimmeh' inside list item.", self.current_tok);
645                std::process::exit(1);
646            }
647        } else if self.is_lemmesee(&self.current_tok) {
648            self.parse_variable_use();
649        } else if self.is_text(&self.current_tok) && !self.current_tok.is_empty() {
650            self.parse_text();
651        }
652        // else empty — fall through to #oic
653
654        if self.is_oic(&self.current_tok) {
655            self.add_str(self.current_tok.clone());
656            self.next_token();
657        } else {
658            eprintln!("Syntax error: '{}' was found when '#oic' was expected.", self.current_tok);
659            std::process::exit(1);
660        }
661        self.close_list();
662    }
663
664    /// <link> ::= LINX ADDRESS OIC
665    /// Called with current_tok == "linx" (caller already consumed #gimmeh).
666    fn parse_link(&mut self) {
667        self.open_list();
668        if self.is_linx(&self.current_tok) {
669            self.add_str(self.current_tok.clone());
670            self.next_token();
671        } else {
672            eprintln!("Syntax error: '{}' was found when 'linx' was expected.", self.current_tok);
673            std::process::exit(1);
674        }
675
676        // ADDRESS ::= text without spaces
677        if !self.is_address(&self.current_tok) {
678            eprintln!("Syntax error: '{}' was found when an address (no spaces) was expected.", self.current_tok);
679            std::process::exit(1);
680        }
681        self.add_str(self.current_tok.clone()); // ADDRESS
682        self.next_token();
683
684        if self.is_oic(&self.current_tok) {
685            self.add_str(self.current_tok.clone());
686            self.next_token();
687        } else {
688            eprintln!("Syntax error: '{}' was found when '#oic' was expected.", self.current_tok);
689            std::process::exit(1);
690        }
691        self.close_list();
692    }
693
694    /// <newline> ::= NEWLINE
695    /// Called with current_tok == "newline" (caller already consumed #gimmeh).
696    fn parse_newline(&mut self) {
697        self.open_list();
698        if self.is_newline(&self.current_tok) {
699            self.add_str(self.current_tok.clone());
700            self.next_token();
701        } else {
702            eprintln!("Syntax error: '{}' was found when 'newline' was expected.", self.current_tok);
703            std::process::exit(1);
704        }
705        self.close_list();
706    }
707
708    /// TEXT ::= plain text (letters, digits, punctuation, spaces)
709    /// Collects consecutive non-keyword text tokens.
710    fn parse_text(&mut self) {
711        while !self.current_tok.is_empty() && self.is_text(&self.current_tok) {
712            self.add_str(self.current_tok.clone());
713            self.next_token();
714        }
715    }
