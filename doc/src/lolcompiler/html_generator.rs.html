1use crate::syntax_analyzer::Node;
2
3// ===================== HtmlGenerator =====================
4
5/// the Html generator takes a resolved syntax tree and walks through it to generate the html
6pub struct HtmlGenerator;
7
8impl HtmlGenerator {
9    pub fn new() -> Self {
10        Self
11    }
12
13    /// Generates an HTML string from the resolved syntax tree.
14    /// It converts the tree to tokens that can then generate the html
15    pub fn generate(&self, tree: &[Node]) -> String {
16        let mut tokens = Vec::new();
17        self.tree_to_tokens(tree, &mut tokens);
18        self.tokens_to_html(&tokens)
19    }
20
21    /// Recursively walks the Node tree and collects all string tokens into a flat list.
22    /// check if current spot is a leaf(token) or branch(array) if leaf copys to token list if branch steps in and call recursively
23    fn tree_to_tokens(&self, nodes: &[Node], out: &mut Vec<String>) {
24        for node in nodes {
25            match node {
26                Node::Str(s)      => out.push(s.clone()),
27                Node::List(items) => self.tree_to_tokens(items, out),
28            }
29        }
30    }
31
32    /// Walks the token list linearly, using a closing-tag stack to produce HTML. 
33    fn tokens_to_html(&self, tokens: &[String]) -> String {
34        let mut out = String::new(); // html data
35        out.push_str("<!DOCTYPE html>\n<html>\n"); // starting html tag
36
37        let mut i = 0;
38        let mut stack: Vec<String> = Vec::new(); // used to collected opened tags and pop them when closing
39        let mut body_open = false; // used to close body at end of token if opened
40
41        while i < tokens.len() { // iter until there are no tokens left
42            let lc = tokens[i].to_ascii_lowercase(); // makes sure token is lowercase 
43
44            match lc.as_str() {
45                // program delimiters — no HTML output
46                // html tag is already opened when initalizing doc
47                // closing body and html is handle outside of loop
48                "#hai" | "#kbye" => {}
49
50                // comment block — collect text and outpots as HTML comment
51                "#obtw" => {
52                    i += 1; // steps to next token
53                    let mut comment_words = Vec::new();
54                    while i < tokens.len() && tokens[i].to_ascii_lowercase() != "#tldr" {
55                        comment_words.push(tokens[i].as_str());
56                        i += 1; // steps to next token
57                    }
58                    out.push_str(&format!("<!-- {} -->\n", comment_words.join(" ")));
59                    // i now points at #tldr; outer i += 1 steps past it
60                }
61
62                "#tldr" => {} // end-of-comment marker, consumed by #obtw loop above
63
64                // close the current tag by popping the stack
65                "#mkay" | "#oic" => {
66                    if let Some(closing) = stack.pop() {
67                        out.push_str(&closing);
68                        // after </head>, open <body>
69                        if closing.starts_with("</head>") && !body_open {
70                            out.push_str("<body>\n");
71                            body_open = true;
72                        }
73                    }
74                }
75
76                // #maek <keyword> ... #mkay
77                "#maek" => {
78                    i += 1; // steps to next token
79                    if i < tokens.len() {
80                        // pushes correct tag for maek <keyword> tag or if 
81                        // also if first tag then push body (only when no head) then correct tag
82                        // pushes closing tag to stack
83                        match tokens[i].to_ascii_lowercase().as_str() {
84                            "head" => {
85                                out.push_str("<head>");
86                                stack.push("</head>\n".to_string());
87                            }
88                            "paragraf" => {
89                                if !body_open { out.push_str("<body>\n"); body_open = true; }
90                                out.push_str("<p>");
91                                stack.push("</p>\n".to_string());
92                            }
93                            "list" => {
94                                if !body_open { out.push_str("<body>\n"); body_open = true; }
95                                out.push_str("<ul>\n");
96                                stack.push("</ul>\n".to_string());
97                            }
98                            _ => { stack.push(String::new()); } // catch all pushes nothing but shouldnt happen 
99                        }
100                    }
101                }
102
103                // #gimmeh <keyword> ... #oic
104                "#gimmeh" => {
105                    i += 1; // steps to next token
106                    if i < tokens.len() {
107                        // matches correct tag and pushes open tag and pushes closing tag to stack
108                        // if newline or linx self closes tag and steps pass #oic since tag already closed
109                        match tokens[i].to_ascii_lowercase().as_str() {
110                            "title"   => { out.push_str("<title>");  stack.push("</title>".to_string()); }
111                            "item"    => { out.push_str("<li>");     stack.push("</li>\n".to_string()); }
112                            "bold"    => { out.push_str("<b>");      stack.push("</b>".to_string()); }
113                            "italics" => { out.push_str("<i>");      stack.push("</i>".to_string()); }
114                            "newline" => {
115                                out.push_str("<br>\n");
116                                stack.push(String::new()); // #oic pops this harmlessly
117                            }
118                            "linx" => {
119                                i += 1; // URL token
120                                let url = if i < tokens.len() { tokens[i].as_str() } else { "" };
121                                out.push_str(&format!("<a href=\"{}\">{}</a>", url, url));
122                                i += 1; // consume the #oic — no stack push needed
123                            }
124                            _ => { stack.push(String::new()); } // catch all pushes nothing but shouldnt happen 
125                        }
126                    }
127                }
128
129                // plain text token
130                // pushes text and space, issue with text adding space to before end tag, not sure how to resolve cleanly
131                _ => {
132                    out.push_str(&tokens[i]);
133                    out.push(' ');
134                }
135            }
136
137            i += 1; // steps to next token
138        }
139        // pushes body close if body open
140        // pushs body open/close if body never opened
141        if body_open {
142            out.push_str("</body>\n");
143        } else {
144            out.push_str("<body>\n</body>\n");
145        }
146        out.push_str("</html>\n"); // closes html tag
147        //returns html data
148        out
149    }
150}
