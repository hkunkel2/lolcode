1/// Trait for a simple lexical analyzer.
2/// Implements a character-by-character analysis
3/// from a state machine design.
4pub trait LexicalAnalyzer {
5    /// Return the next character from the input.
6    /// If input is exhausted, should terminate the program.
7    fn get_char(&mut self) -> char;
8
9    /// Add a character to the current potential token.
10    fn add_char(&mut self, c: char);
11
12    /// Lookup a potential token to determine if it is valid.
13    /// Returns true if a valid token/lexeme, false otherwise.
14    fn lookup(&self, s: &str) -> bool;
15}

50/// LolcodeLexicalAnalyzer 
51impl LolcodeLexicalAnalyzer {
52    /// creates lexer defines keywords and valid tokens 
53    pub fn new(source: &str) -> Self {
54        Self {
55            input: source.chars().collect(),
56            position: 0,
57            current_build: String::new(),
58            tokens: Vec::new(),
59            hai: vec!["#hai".into()],
60            kbye: vec!["#kbye".into()],
61            obtw: vec!["#obtw".into()],
62            tldr: vec!["#tldr".into()],
63            maek: vec!["#maek".into()],
64            oic: vec!["#oic".into()],
65            gimmeh: vec!["#gimmeh".into()],
66            mkay: vec!["#mkay".into()],
67            head: vec!["head".into()],
68            title: vec!["title".into()],
69            paragraf: vec!["paragraf".into()],
70            bold: vec!["bold".into()],
71            italics: vec!["italics".into()],
72            list: vec!["list".into()],
73            item: vec!["item".into()],
74            newline: vec!["newline".into()],
75            linx: vec!["linx".into()],
76            ihaz: vec!["#ihaz".into()],
77            itiz: vec!["#itiz".into()],
78            lemmesee: vec!["#lemmesee".into()],
79            var_name: Vec::new(),                                                                                   
80            var_value: Vec::new(),                                                                                  
81            text: Vec::new(),                                                                                       
82            address: Vec::new(),
83        }
84    }
85
86    /// Check if a character is an allowed.
87    fn is_allowed_char(c: char) -> bool {
88        c.is_ascii_alphanumeric()
89            || matches!(c, ',' | '.' | '"' | '\'' | ':' | '?' | '!' | '%' | '/' | '=' | '\t' | '\n' | ' ')
90    }
91
92    /// VARNAME ::= any single word (A–Z, a–z, no spaces)
93    pub fn is_var_name(s: &str) -> bool {
94        !s.is_empty() && s.chars().all(|c| c.is_ascii_alphabetic())
95    }
96
97    /// VARVALUE ::= allowed text characters (no spaces)
98    pub fn is_var_value(s: &str) -> bool {
99        !s.is_empty() && s.chars().all(|c| Self::is_allowed_char(c) && c != ' ')
100    }
101
102    /// TEXT ::= plain text (allowed characters including spaces) *empty allowed
103    pub fn is_text(s: &str) -> bool {
104        s.chars().all(|c| Self::is_allowed_char(c))
105    }
106
107    /// ADDRESS ::= text without spaces
108    pub fn is_address(s: &str) -> bool {
109        !s.is_empty() && s.chars().all(|c| Self::is_allowed_char(c) && c != ' ')
110    }
111    /// steps char by char and on ws creates token
112    pub fn tokenize(&mut self) {
113        loop {
114            let c = self.get_char();
115            if c == '\0' {
116                break;
117            }
118            if c.is_whitespace() {
119                if !self.current_build.is_empty() {
120                    self.tokens.push(std::mem::take(&mut self.current_build));
121                }
122            } else {
123                self.add_char(c);
124            }
125        }
126        if !self.current_build.is_empty() {
127            self.tokens.push(std::mem::take(&mut self.current_build));
128        }
129        self.tokens.reverse();
130    }
131}

134/// from simple lexer in lab 4 from string it gets char then adds char to token and lookups token against valid tokens 
135impl LexicalAnalyzer for LolcodeLexicalAnalyzer {
136    /// gets char from string input
137    fn get_char(&mut self) -> char {
138        if self.position < self.input.len() {
139            let c = self.input[self.position];
140            self.position += 1;
141            c
142        } else {
143            '\0'
144        }
145    }
146
147    /// adds char to token 
148    fn add_char(&mut self, c: char) {
149        self.current_build.push(c);
150    }
151
152    /// checks if token is valid in the lookup table(predefined tokens and chars)
153    fn lookup(&self, s: &str) -> bool {
154        let lower = s.to_ascii_lowercase();
155        self.hai.iter().any(|t| t == &lower)
156            || self.kbye.iter().any(|t| t == &lower)
157            || self.obtw.iter().any(|t| t == &lower)
158            || self.tldr.iter().any(|t| t == &lower)
159            || self.maek.iter().any(|t| t == &lower)
160            || self.oic.iter().any(|t| t == &lower)
161            || self.gimmeh.iter().any(|t| t == &lower)
162            || self.mkay.iter().any(|t| t == &lower)
163            || self.head.iter().any(|t| t == &lower)
164            || self.title.iter().any(|t| t == &lower)
165            || self.paragraf.iter().any(|t| t == &lower)
166            || self.bold.iter().any(|t| t == &lower)
167            || self.italics.iter().any(|t| t == &lower)
168            || self.list.iter().any(|t| t == &lower)
169            || self.item.iter().any(|t| t == &lower)
170            || self.newline.iter().any(|t| t == &lower)
171            || self.linx.iter().any(|t| t == &lower)
172            || self.ihaz.iter().any(|t| t == &lower)
173            || self.itiz.iter().any(|t| t == &lower)
174            || self.lemmesee.iter().any(|t| t == &lower)
175            || Self::is_var_name(&lower)                                                                                 
176            || Self::is_var_value(&lower)                                                                                
177            || Self::is_text(&lower)                                                                                     
178            || Self::is_address(&lower)
179    }
